### バック
```Rust
use std::env;

use actix_web::{App, HttpResponse, HttpServer, Responder, error::ErrorInternalServerError, mime, web};
use serde::{Deserialize, Serialize};

#[actix_web::main]
async fn main() -> Result<(), std::io::Error> {
	let host = env::var("SERVER_HOST").expect("`SERVER_HOST` is undefined");
	let port = env::var("SERVER_PORT").expect("`SERVER_PORT` is undefined");

	HttpServer::new(|| App::new().route("/", web::to(index)).route("register", web::post().to(register)))
		.bind(format!("{host}:{port}"))?
		.run()
		.await
}

async fn index() -> impl Responder {
	HttpResponse::Ok()
		.content_type(mime::TEXT_HTML)
		.body(std::fs::read_to_string("app/colllus/resource/html/index.html").unwrap())
}

#[derive(Deserialize)]
struct Register {
	code: String,
	name: String,
}
async fn register(info: web::Form<Register>) -> Result<impl Responder, actix_web::Error> {
	#[derive(Serialize)]
	struct Auth {
		code: String,
	}
	let client = reqwest::Client::new();
	// 外部から通信するならここを"https://untroche.com/auth"とかにする　dockerなしの内部通信なら"http://localhost:8000/auth"
	let id = match client
		.post("http://portal:8000/auth")
		.json(&Auth { code: info.code.clone() })
		.send()
		.await
		.and_then(|res| res.error_for_status())
	{
		Ok(ok) => ok.text().await.map_err(|err| ErrorInternalServerError(err))?,
		Err(err) => {
			eprintln!("{err:?}");
			return Err(ErrorInternalServerError(err));
		}
	};
	Ok(format!("{id}: {}", info.name))
}
```
